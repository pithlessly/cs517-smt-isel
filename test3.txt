[program]

a = load()
b = a
c = a
2 = a
4 = a
discriminant = sub(mul(b, b), mul(mul(4, a), c))
num1 = add(neg(b), sqrt(discriminant))
num2 = sub(neg(b), sqrt(discriminant))

r2 = div(num2, mul(2, a))
program = pair(r1, r2)
